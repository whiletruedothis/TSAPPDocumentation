import { Component, OnInit } from '@angular/core';
import {User} from '../../../shared/entities/user';
import {Role} from '../../../shared/entities/role';
import {FormBuilder, FormGroup, Validators, FormControl} from '@angular/forms';
import { UserService } from './../../../core/services/user.service';
import { Router } from '@angular/router';
import {Observable} from 'rxjs';
import {startWith, map} from 'rxjs/operators';
import { Career } from '../../../shared/interfaces/carrer';
import { Constants } from '../../../shared/util/constants';
import { _filter } from '../../../shared/util/commons';
import { MatSnackBar } from '@angular/material';
import { SuccessMessageToastComponent } from 'app/pages/success-message-toast/success-message-toast.component';


@Component({
  selector: 'app-form-account',
  templateUrl: './form-account.component.html',
  styleUrls: ['./form-account.component.scss']
})
export class FormAccountComponent implements OnInit {

  user: User;
  users: Partial<User>[];
  role: Role;
  roleName: string;
  formAccount: FormGroup;
  formRole: FormGroup;
  careerOptions: Observable<Career[]>;
  careerGroups: Career[] = Constants.CARRERS;
  telephone: number;

  constructor(private router: Router,
              private userService: UserService,
              private fb: FormBuilder,
              public snackBar: MatSnackBar) {
    this.createFormUserAndRole();
    this.fb = new FormBuilder();
  }

  ngOnInit() {
    this.createFormUserAndRole();
    this.careerOptions = this.formAccount.get('Career')!.valueChanges
    .pipe(
      startWith(''),
      map(value => this._filterGroup(value))
    );

  }

  createFormUserAndRole() {

    this.formAccount = this.fb.group({
      firstname: ['', Validators.required],
      lastname: ['', Validators.required],
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(8)]],
      password2: ['', [Validators.required, Validators.minLength(8)]],
      Career: [null],
      telephone: [null]
    }, {
      validator: this.matchingPasswords('password', 'password2')
     });

    this.formRole = this.fb.group({
      name: ['', Validators.required]
    });
  }

  get f() {
     return this.formAccount.controls; 
  }

  get fr() { 
    return this.formRole.controls;
  }

  matchingPasswords(passwordKey: string, password2Key: string) {
    return (group: FormGroup): {[key: string]: any} => {
      const password = group.controls[passwordKey];
      const password2 = group.controls[password2Key];

      if (password.value !== password2.value) {
        return {
          mismatchedPasswords: true
        };
      }
    };
  }

  selectedRole(event: any) {
    return this.roleName = event.value;
  }

  onSubmit(formAccount: any, formRole: any) {
    if (formAccount.telephone === '') {
      formAccount.telephone = null;
    }
    this.telephone = formAccount.telephone;
    this.user = new User(
      formAccount.firstname,
      formAccount.lastname,
      formAccount.email,
      formAccount.password,
      formAccount.Career,
      formAccount.telephone);
    this.role = new Role(formRole.name);
    this.userService.saveUserAsAdmin(this.user, this.role.role)
    .subscribe(response => {
      this.openSnackBar();
      this.router.navigate(['seeAccounts']);
    });
  }

  openSnackBar() {
    this.snackBar.openFromComponent(SuccessMessageToastComponent, {
      data: 'Usuario creado exitosamente ✔️',
      duration: 2500
    });
  }

  private _filterGroup(value: string): Career[] {
    if (value) {
      return this.careerGroups
        .map(group => ({area: group.area, careerName: _filter(group.careerName, value)}))
        .filter(group => group.careerName.length > 0);
    }
    return this.careerGroups;
  }

}
