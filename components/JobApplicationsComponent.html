import { Component, OnInit } from '@angular/core';
import { GlobalPlan } from '@shared/entities/global-plan';
import { MatTableDataSource } from '@angular/material';
import { GlobalPlanService } from '@core/services/global-plan.service';
import { Router } from '@angular/router';
import { AuthService } from '@core/services/auth.service';

@Component({
  selector: 'app-job-applications',
  templateUrl: './job-applications.component.html',
  styleUrls: ['./job-applications.component.scss']
})
export class JobApplicationsComponent implements OnInit {
  globalPlans: GlobalPlan[];


  displayedColumns: string[] = ['title', 'filename', 'file'];
  dataSource = new MatTableDataSource(this.globalPlans);

  email: string = this.authService.email;

  constructor(private globalPlanService: GlobalPlanService,
              private authService: AuthService,
              private router: Router ) { }

  ngOnInit() {
    this.getTeacherGlobalPlans();
  }

   getTeacherGlobalPlans() {
   this.globalPlanService.getTeacherGlobalPlans(this.email)
    .subscribe(response => {
      this.globalPlans = response;
      this.dataSource = new MatTableDataSource(this.globalPlans);
    });

   }

   downloadGlobalPlan(event, element) {
   window.open('https://teacherselection.herokuapp.com/api/globalPlan?email=' + element.email + '&title=' + element.title);

  }

}
